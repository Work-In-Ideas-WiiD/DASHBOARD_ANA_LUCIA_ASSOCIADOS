const formSchema = zod.object({

export function AdministradoresTable() {

            search: '',

    });
 
    useEffect(() => {
        getData(page, getValues("search"));

    async function getData(pageParam: number, likeParam: string = "") {
        try {
            setFetching(true);

            setPages(data.from);
            setAdmins(data.data);
            setNoContent(data.data.length == 0);

            setFetching(false);
